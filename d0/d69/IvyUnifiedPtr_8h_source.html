255 template<typename T, IvyPointerType IPT> __HOST_DEVICE__ void IvyUnifiedPtr<T, IPT>::release(){
256 if (cblock_){
257 // Atomically decrement first. The thread that observes the transition to zero
258 // (i.e. sees a previous value of 1) is the sole owner responsible for cleanup.
259 // This avoids the check-then-act race of reading use_count() and then freeing.
260 auto const prev_count = this->inc_dec_counter(false);
261 if (prev_count==1){
262 // We were the only owner, so we can clean up. We must have write access first.
263 if (this->check_write_access()){
264 // Sole owner: snapshot the whole control block once (no concurrent refcount activity).
265 control_block_type const cb = this->load_control_block();
266 operate_with_GPU_stream_from_pointer(
267 stream_, ref_stream,
268 __ENCAPSULATE__(
269 if (ptr_){
270 element_allocator_traits::destruct(ptr_, cb.size, cb.mem_type, ref_stream);
271 element_allocator_traits::deallocate(ptr_, cb.capacity, cb.mem_type, ref_stream);
272 }
273 control_block_allocator_traits::destroy(cblock_, 1, exec_mem_type_, ref_stream);
274 )
275 );
276 }
277 else{

470 template<typename T, IvyPointerType IPT> __HOST_DEVICE__ bool IvyUnifiedPtr<T, IPT>::transfer_impl(IvyMemoryType const& new_mem_type, bool transfer_all, bool copy_ptr, bool release_old){
471 bool res = true;
472 if (ptr_){
473 auto const current_mem_type = this->get_memory_type();
474 if (copy_ptr || current_mem_type != new_mem_type){
475 IvyMemoryType misc_mem_type = (transfer_all ? new_mem_type : exec_mem_type_);
476 pointer new_ptr_ = nullptr;
477 control_block_type* new_cblock_ = nullptr;
478 size_type const n_size = this->size();
479 size_type const n_capacity = this->capacity();
480
481 // The metadata is now a single coalesced control block, so a transfer allocates and
482 // moves one object instead of four. Running this serially is faster than spawning
483 // OpenMP sections and avoids the associated leak/race surface; OpenMP is reserved
484 // for element-array loops.
485 operate_with_GPU_stream_from_pointer(
486 stream_, ref_stream,
487 __ENCAPSULATE__(
488 res &= element_allocator_traits::allocate(new_ptr_, n_capacity, new_mem_type, ref_stream);
489 res &= element_allocator_traits::transfer(new_ptr_, ptr_, n_size, new_mem_type, current_mem_type, ref_stream);
490 new_cblock_ = control_block_allocator_traits::allocate(1, misc_mem_type, ref_stream);
491 res &= (new_cblock_ != nullptr);
492 )
493 );
494
495 // The following line is correct.
496 // We should never release when a copy is being made.
497 // A copy is a direct memcpy operation from a read-only source.
498 if (!copy_ptr || release_old) this->release();
499
500 // If we were not the progenitor before, release() will abort the program, and execution will never reach beyond this point.
501 // Otherwise, at this point, we have released the old memory successfully if we needed to.
502 // For that reason, we can assume a new progenitor role now.
503 progenitor_mem_type_ = IvyMemoryHelpers::get_execution_default_memory();
504
505 exec_mem_type_ = misc_mem_type;
506 ptr_ = new_ptr_;
507 cblock_ = new_cblock_;
508 // Populate the freshly allocated control block (refcount starts at 1 for the new owner).
509 control_block_type const cb{ __STATIC_CAST__(counter_type, 1), n_size, n_capacity, new_mem_type };
510 this->store_control_block(cb);
511 }
512 }
513 return res;
514 }
515 template<typename T, IvyPointerType IPT> __HOST__ bool IvyUnifiedPtr<T, IPT>::transfer(IvyMemoryType const& new_mem_type, bool transfer_all, bool release_old){
516 return this->transfer_impl(new_mem_type, transfer_all, false, release_old);

579 template<typename T, IvyPointerType IPT> __HOST_DEVICE__ IvyUnifiedPtr<T, IPT>::counter_type IvyUnifiedPtr<T, IPT>::use_count() const{
580 if (!cblock_) return 0;
581 if (this->control_block_is_addressable()){
582 return std_atomic::atomic_ref<counter_type>(cblock_->ref_count).load(std_atomic::memory_order_acquire);
583 }
584 else return this->read_meta_field(&cblock_->ref_count);
585 }
586 template<typename T, IvyPointerType IPT> __HOST_DEVICE__ IvyUnifiedPtr<T, IPT>::counter_type IvyUnifiedPtr<T, IPT>::inc_dec_counter(bool do_inc){
587 if (!cblock_){
588 __PRINT_ERROR__("IvyUnifiedPtr::inc_dec_counter() failed: cblock_ is null.\n");
589 assert(false);
590 }
591 if (this->control_block_is_addressable()){
592 // The control block lives in directly-addressable memory of the current execution space.
593 // Mutate the counter atomically so concurrent copies/destructions of shared_ptr are race-free.
594 // Increments may use relaxed ordering; the decrement must be acquire/release so that
595 // the thread observing the transition to zero sees all prior writes before freeing.
596 // std_atomic::atomic_ref maps to cuda::std::atomic_ref (CUDA) or std::atomic_ref (host),
597 // so both execution spaces share the same ordering guarantees.
598 std_atomic::atomic_ref<counter_type> a_ref_count(cblock_->ref_count);
599 if (do_inc) return a_ref_count.fetch_add(__STATIC_CAST__(counter_type, 1), std_atomic::memory_order_relaxed);
600 else return a_ref_count.fetch_sub(__STATIC_CAST__(counter_type, 1), std_atomic::memory_order_acq_rel);
601 }
602 else{
603 // Non-addressable (e.g. device) memory reached from another execution space: the counter
604 // cannot be mutated in place, so read just the ref_count field into an on-stack temporary,
605 // modify it, and write it back. Touching only this field avoids clobbering concurrent
606 // updates to the other control-block fields.
607 // A host-side hardware atomic cannot operate on memory that is not directly addressable
608 // from the host (the counter must be staged through a transfer), so we instead serialize
609 // the staged read-modify-write with a host-side spinlock keyed by the control-block
610 // address. This makes concurrent host-thread ownership updates race-free; only host threads
611 // can reach this branch, since device code addresses its own memory directly and takes the
612 // hardware-atomic path above.
613#if DEVICE_CODE == DEVICE_CODE_HOST
614 std_atomic::atomic_flag& rc_lock = detail::nonaddressable_refcount_lock(cblock_);
615 while (rc_lock.test_and_set(std_atomic::memory_order_acquire)){ ::std::this_thread::yield(); /* spin with backoff until acquired */ }
616#endif
617 counter_type prev = this->read_meta_field(&cblock_->ref_count);
618 counter_type next = prev;
619 if (do_inc) ++next;
620 else --next;
621 this->write_meta_field(&cblock_->ref_count, next);
622#if DEVICE_CODE == DEVICE_CODE_HOST
623 rc_lock.clear(std_atomic::memory_order_release);
624#endif
625 return prev;
626 }
627 }
628 template<typename T, IvyPointerType IPT> __HOST_DEVICE__ void IvyUnifiedPtr<T, IPT>::inc_dec_size(bool do_inc){

631 assert(false);
